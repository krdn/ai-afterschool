    ```typescript
    }
    ```
    
    기능:

       - variant: "ghost" (헤더 스타일에 맞춤)
       - size: "sm"
       - 아이콘: Flag 또는 AlertCircle (lucide-react)
       - 텍스트: "이슈 보고" (모바일에서는 아이콘만)
    2. 클릭 시 모달 열기
       - isOpen 상태: true
       - IssueReportModal 렌더링
    3. 모달 닫기 처리
       - onClose: isOpen 상태 false
    
    구현:
    ```tsx
    'use client'
    
    export function IssueReportButton({ userRole }: IssueReportButtonProps) {
       const [isOpen, setIsOpen] = useState(false);
       
       return (
         <>
           <Button 
             variant="ghost" 
             size="sm"
             onClick={() => setIsOpen(true)}
             className="hidden md:flex"
           >
             <Flag className="h-4 w-4 mr-2" />
             이슈 보고
           </Button>
           <Button
             variant="ghost"
             size="icon"
             onClick={() => setIsOpen(true)}
             className="md:hidden"
           >
             <Flag className="h-5 w-5" />
           </Button>
           <IssueReportModal 
             isOpen={isOpen} 
             onClose={() => setIsOpen(false)}
             userRole={userRole}
           />
         </>
       );
    }
    ```
