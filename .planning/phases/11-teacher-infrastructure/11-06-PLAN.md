    ```typescript
    import { notFound } from 'next/navigation'
    import { verifySession } from '@/lib/dal'
    import { getTeacherById } from '@/lib/actions/teachers'
    import { TeacherDetail } from '@/components/teachers/teacher-detail'
    import { Button } from '@/components/ui/button'
    import Link from 'next/link'
    import { ArrowLeft } from 'lucide-react'

    export default async function TeacherDetailPage({
      params,
    }: {
      params: { id: string }
    }) {
      const session = await verifySession()
      const teacher = await getTeacherById(params.id)

      if (!teacher) {
        notFound()
      }

      return (
        <div className="space-y-6">
          <div className="flex items-center gap-4">
            <Button variant="ghost" size="sm" asChild>
              <Link href="/teachers">
                <ArrowLeft className="h-4 w-4 mr-2" />
                목록으로
              </Link>
            </Button>
            <div>
              <h1 className="text-2xl font-bold">{teacher.name}</h1>
              <p className="text-gray-500">선생님 상세 정보</p>
            </div>
          </div>

          <TeacherDetail teacher={teacher} currentRole={session.role} />
        </div>
      )
    }
    ```

    주요 기능:
    1. 동적 라우팅: [id] 파라미터로 선생님 ID 받음
    2. getTeacherById 호출로 선생님 상세 조회 (RBAC 권한 검증 포함)
    3. 선생님 없으면 404 페이지 (notFound())
    4. 뒤로 가기 버튼 (목록으로)
    5. TeacherDetail 컴포넌트로 정보 표시

    주의: getTeacherById 함수 내부에서 이미 RBAC 권한 검증을 수행하므로 페이지에서 추가 검증 불필요
