useEffect(() => {
  if (isOpen && !data) {
    setIsLoading(true)
    getStudentAISupportDataAction(studentId)

      })
      .finally(() => setIsLoading(false))
  }

궁합 점수가 없을 때 자동 계산 기능을 연동한다.

handleCalculateCompatibility 구현:
```typescript
const handleCalculateCompatibility = async () => {
  setIsCalculating(true)
  try {
    // Phase 13 calculateCompatibilityAction 호출
    const result = await calculateCompatibilityAction({
      teacherId: session.userId,  // 현재 로그인한 선생님
      studentId,
    })

    if (result.success) {
      // 데이터 갱신
      setData(prev => prev ? {
        ...prev,
        compatibility: result.data.compatibility,
      } : null)
      toast.success("궁합 점수가 계산되었습니다")
    } else {
      toast.error(result.error || "궁합 점수 계산에 실패했습니다")
    }
  } catch (error) {
    toast.error("궁합 점수 계산 중 오류가 발생했습니다")
  } finally {
    setIsCalculating(false)
  }
}
```

CONTEXT.md 결정사항 반영:
- "궁합 정보가 아직 계산되지 않은 경우 상담 화면 진입 시 필요한 데이터가 있으면 자동 계산"
- canCalculateCompatibility가 true이고 compatibility가 null이면 자동 계산 트리거

자동 계산 로직 (useEffect 추가):
```typescript
useEffect(() => {
  if (data && !data.compatibility && data.canCalculateCompatibility) {
    handleCalculateCompatibility()
  }
}, [data?.canCalculateCompatibility, data?.compatibility])
```
