    3. 데이터 페칭 (useEffect):
       ```typescript
       // 컴포넌트 마운트 시 학생 목록 조회
       useEffect(() => {
         const fetchStudents = async () => {
           const result = await getStudentsAction({})
           if (result.success && result.data) {
             setStudents(result.data)
           }
         }
         fetchStudents()
       }, [])

       // 날짜 선택 시 해당 날의 예약된 슬롯 조회
       useEffect(() => {
         if (selectedDate) {
           const fetchReservedSlots = async () => {
             const dayStart = new Date(selectedDate)
             dayStart.setHours(0, 0, 0, 0)
             const dayEnd = new Date(selectedDate)
             dayEnd.setHours(23, 59, 59, 999)

             const result = await getReservationsAction({
               startDate: dayStart,
               endDate: dayEnd,
               status: undefined // 전체 상태 (취소/노쇼 포함)
             })

             if (result.success && result.data) {
               // 예약된 시간 슬롯 추출 (HH:mm 형식)
               const slots = result.data
                 .filter(r => r.status !== 'CANCELLED' && r.status !== 'NO_SHOW')
                 .map(r => {
                   const date = new Date(r.scheduledAt)
                   return `${String(date.getHours()).padStart(2, '0')}:${String(date.getMinutes()).padStart(2, '0')}`
                 })
               setReservedSlots(slots)
             }
           }
           fetchReservedSlots()
         } else {
           setReservedSlots([])
         }
       }, [selectedDate])
       ```

    8. 제출 로직:
       ```typescript
       const handleSubmit = async () => {
         // 폼 검증
         if (!selectedDate || !selectedTime || !selectedStudentId || !selectedParentId || !topic.trim()) {
           toast.error('모든 필드를 입력해주세요.')
           return
         }

         // scheduledAt DateTime 생성
         const [hours, minutes] = selectedTime.split(':').map(Number)
         const scheduledAt = new Date(selectedDate)
         scheduledAt.setHours(hours, minutes, 0, 0)

         setIsSubmitting(true)
         try {
           const result = await createReservationAction({
             scheduledAt: scheduledAt.toISOString(),
             studentId: selectedStudentId,
             parentId: selectedParentId,
             topic: topic.trim()
           })

           if (result.success) {
             toast.success('예약이 등록되었습니다.')
             onSuccess()
           } else {
             toast.error(result.error || '예약 등록에 실패했습니다.')
           }
         } catch (error) {
           toast.error('예약 등록 중 오류가 발생했습니다.')
         } finally {
           setIsSubmitting(false)
         }
       }
       ```
