<task type="auto">
  <name>Task 1: Create database backup script</name>
  <files>scripts/backup-db.sh</files>
  <action>
    Create scripts/backup-db.sh:
    - Add shebang: `#!/bin/bash`
    - Set strict mode: `set -euo pipefail`
    - Define environment variables with defaults:
      - BACKUP_DIR="${BACKUP_DIR:-./backups}"
      - DB_NAME="${DB_NAME:-ai_afterschool}"
      - DB_USER="${DB_USER:-postgres}"
      - DB_HOST="${DB_HOST:-localhost}"
      - DB_PORT="${DB_PORT:-5432}"
      - RETENTION_DAYS="${RETENTION_DAYS:-30}"
    - Generate timestamp: `DATE=$(date +%Y-%m-%d_%H-%M-%S)`
    - Define backup filename: `BACKUP_FILE="$BACKUP_DIR/${DB_NAME}-${DATE}.sql.gz"`
    - Create backup directory: `mkdir -p "$BACKUP_DIR"`
    - Run pg_dump with compression: `docker exec postgres pg_dump -U "$DB_USER" "$DB_NAME" | gzip > "$BACKUP_FILE"`
    - Check exit status and log success/failure
    - Clean up old backups: `find "$BACKUP_DIR" -name "${DB_NAME}-*.sql.gz" -mtime +$RETENTION_DAYS -delete`
    - Verify database connectivity: `docker exec postgres psql -U "$DB_USER" -d postgres -c "SELECT 1"`
    - Log all operations to stdout/stderr for cron to capture
  </action>
  <verify>`test -f scripts/backup-db.sh && grep -E "pg_dump|gzip|RETENTION_DAYS" scripts/backup-db.sh`</verify>
  <done>Backup script created with pg_dump, gzip compression, and retention policy.</done>
