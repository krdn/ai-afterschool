import { db } from '@/lib/db';
import { COST_PER_MILLION_TOKENS, PROVIDER_CONFIGS, type ProviderName, type FeatureType } from './providers';
import { getEnabledProviders } from './config';
import { getCurrentPeriodCost } from './usage-tracker';

// ============================================================
// Cost-Based Provider Ordering
// ============================================================

/**
 * 비용 순으로 제공자를 정렬합니다 (저렴한 것 우선).
 */
export function sortProvidersByCost(providers: ProviderName[]): ProviderName[] {
  return [...providers].sort((a, b) => {
    const costA = COST_PER_MILLION_TOKENS[a].input + COST_PER_MILLION_TOKENS[a].output;
    const costB = COST_PER_MILLION_TOKENS[b].input + COST_PER_MILLION_TOKENS[b].output;
    return costA - costB;
  });
}

/**
 * 비전 기능 지원 여부로 제공자를 필터링합니다.
 */
export function filterVisionProviders(providers: ProviderName[]): ProviderName[] {
  return providers.filter((p) => PROVIDER_CONFIGS[p].supportsVision);
}

/**
 * 기능에 맞게 최적화된 제공자 순서를 반환합니다.
 */
export async function optimizeProviderOrder(
  featureType: FeatureType,
  customOrder?: ProviderName[]
): Promise<ProviderName[]> {
  // 활성화된 제공자만
  const enabledProviders = await getEnabledProviders();
  
  if (enabledProviders.length === 0) {
    throw new Error('No enabled providers available');
  }

  // 사용자 지정 순서가 있으면 활성화된 것만 필터링
  let providers = customOrder
    ? customOrder.filter((p) => enabledProviders.includes(p))
    : enabledProviders;

  // 비전 기능은 비전 지원 제공자만
  const visionFeatures: FeatureType[] = ['face_analysis', 'palm_analysis'];
  if (visionFeatures.includes(featureType)) {
    providers = filterVisionProviders(providers);
    
    if (providers.length === 0) {
      throw new Error('No vision-capable providers enabled');
    }
  }

  // 비용 순 정렬 (기본 최적화)
  return sortProvidersByCost(providers);
}

// ============================================================
// Budget Threshold Checking
// ============================================================

interface BudgetStatus {
  period: 'daily' | 'weekly' | 'monthly';
  budgetUsd: number;
  currentCostUsd: number;
  percentUsed: number;
  threshold80Reached: boolean;
  threshold100Reached: boolean;
  shouldAlert: boolean;
  alertThreshold: 80 | 100 | null;
}

/**
 * 현재 예산 상태를 확인합니다.
 */
export async function checkBudgetStatus(period: 'daily' | 'weekly' | 'monthly'): Promise<BudgetStatus | null> {
  const budget = await db.lLMBudget.findUnique({
    where: { period },
  });

  if (!budget) {
    return null;
  }

  const currentCost = await getCurrentPeriodCost(period);
  const percentUsed = (currentCost / budget.budgetUsd) * 100;
  
  const threshold80Reached = percentUsed >= 80;
  const threshold100Reached = percentUsed >= 100;

  // 알림이 필요한지 확인 (debounce: 마지막 알림 후 1시간)
  const oneHourAgo = new Date(Date.now() - 60 * 60 * 1000);
  const canAlert = !budget.lastAlertAt || budget.lastAlertAt < oneHourAgo;
  
  let shouldAlert = false;
  let alertThreshold: 80 | 100 | null = null;

  if (canAlert) {
    if (threshold100Reached && budget.alertAt100 && budget.lastAlertThreshold !== 100) {
      shouldAlert = true;
      alertThreshold = 100;
    } else if (threshold80Reached && !threshold100Reached && budget.alertAt80 && budget.lastAlertThreshold !== 80) {
      shouldAlert = true;
      alertThreshold = 80;
    }
  }

  return {
    period,
    budgetUsd: budget.budgetUsd,
    currentCostUsd: currentCost,
    percentUsed,
    threshold80Reached,
    threshold100Reached,
    shouldAlert,
    alertThreshold,
  };
}

/**
 * 모든 예산 상태를 확인합니다.
 */
export async function checkAllBudgetStatuses(): Promise<BudgetStatus[]> {
  const periods: Array<'daily' | 'weekly' | 'monthly'> = ['daily', 'weekly', 'monthly'];
  const statuses: BudgetStatus[] = [];

  for (const period of periods) {
    const status = await checkBudgetStatus(period);
    if (status) {
      statuses.push(status);
    }
  }

  return statuses;
}

/**
 * 알림 발송 후 상태를 업데이트합니다.
 */
export async function markAlertSent(period: 'daily' | 'weekly' | 'monthly', threshold: 80 | 100) {
  await db.lLMBudget.update({
    where: { period },
    data: {
      lastAlertAt: new Date(),
      lastAlertThreshold: threshold,
    },
  });
}

/**
 * 기간이 바뀌면 알림 상태를 리셋합니다.
 */
export async function resetAlertStatus(period: 'daily' | 'weekly' | 'monthly') {
  await db.lLMBudget.update({
    where: { period },
    data: {
      lastAlertThreshold: null,
    },
  });
}

// ============================================================
// Estimated Cost Calculation
// ============================================================

/**
 * 요청 예상 비용을 계산합니다 (사전 추정).
 */
export function estimateRequestCost(
  provider: ProviderName,
  estimatedInputTokens: number,
  estimatedOutputTokens: number
): number {
  const costs = COST_PER_MILLION_TOKENS[provider];
  
  const inputCost = (estimatedInputTokens / 1_000_000) * costs.input;
  const outputCost = (estimatedOutputTokens / 1_000_000) * costs.output;

  return inputCost + outputCost;
}

/**
 * 현재 비용 상태 요약을 반환합니다.
 */
export async function getCostSummary(): Promise<{
  daily: { cost: number; budget: number | null; percent: number };
  weekly: { cost: number; budget: number | null; percent: number };
  monthly: { cost: number; budget: number | null; percent: number };
}> {
  const [dailyCost, weeklyCost, monthlyCost] = await Promise.all([
    getCurrentPeriodCost('daily'),
    getCurrentPeriodCost('weekly'),
    getCurrentPeriodCost('monthly'),
  ]);

  const [dailyBudget, weeklyBudget, monthlyBudget] = await Promise.all([
    db.lLMBudget.findUnique({ where: { period: 'daily' } }),
    db.lLMBudget.findUnique({ where: { period: 'weekly' } }),
    db.lLMBudget.findUnique({ where: { period: 'monthly' } }),
  ]);

  return {
    daily: {
      cost: dailyCost,
      budget: dailyBudget?.budgetUsd ?? null,
      percent: dailyBudget ? (dailyCost / dailyBudget.budgetUsd) * 100 : 0,
    },
    weekly: {
      cost: weeklyCost,
      budget: weeklyBudget?.budgetUsd ?? null,
      percent: weeklyBudget ? (weeklyCost / weeklyBudget.budgetUsd) * 100 : 0,
    },
    monthly: {
      cost: monthlyCost,
      budget: monthlyBudget?.budgetUsd ?? null,
      percent: monthlyBudget ? (monthlyCost / monthlyBudget.budgetUsd) * 100 : 0,
    },
  };
}

'use client';

import { useState } from 'react';
import { Card, CardContent, CardDescription, CardHeader, CardTitle } from '@/components/ui/card';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';
import { Label } from '@/components/ui/label';
import { Switch } from '@/components/ui/switch';
import { Loader2, AlertTriangle, DollarSign } from 'lucide-react';
import { Progress } from '@/components/ui/progress';
import { saveBudgetConfigAction } from '@/lib/actions/llm-settings';

interface BudgetSettingsProps {
  budgets: Array<{
    period: 'daily' | 'weekly' | 'monthly';
    budgetUsd: number;
    alertAt80: boolean;
    alertAt100: boolean;
  }>;
  costSummary: {
    daily: { cost: number; budget: number | null; percent: number };
    weekly: { cost: number; budget: number | null; percent: number };
    monthly: { cost: number; budget: number | null; percent: number };
  };
}

const PERIOD_LABELS: Record<string, string> = {
  daily: '일일',
  weekly: '주간',
  monthly: '월간',
};

export function BudgetSettings({ budgets, costSummary }: BudgetSettingsProps) {
  const [configs, setConfigs] = useState(
    ['daily', 'weekly', 'monthly'].map((period) => {
      const existing = budgets.find((b) => b.period === period);
      return {
        period: period as 'daily' | 'weekly' | 'monthly',
        budgetUsd: existing?.budgetUsd ?? 0,
        alertAt80: existing?.alertAt80 ?? true,
        alertAt100: existing?.alertAt100 ?? true,
      };
    })
  );
  const [savingPeriod, setSavingPeriod] = useState<string | null>(null);

  const handleSave = async (period: 'daily' | 'weekly' | 'monthly') => {
    const config = configs.find((c) => c.period === period);
    if (!config) return;

    setSavingPeriod(period);
    try {
      await saveBudgetConfigAction(config);
    } finally {
      setSavingPeriod(null);
    }
  };

  const updateConfig = (period: string, updates: Partial<typeof configs[0]>) => {
    setConfigs((prev) =>
      prev.map((c) => (c.period === period ? { ...c, ...updates } : c))
    );
  };

  return (
    <Card>
      <CardHeader>
        <CardTitle className="flex items-center gap-2">
          <DollarSign className="h-5 w-5" />
          예산 설정
        </CardTitle>
        <CardDescription>
          비용 예산을 설정하고 임계값 도달 시 알림을 받습니다.
        </CardDescription>
      </CardHeader>
      <CardContent className="space-y-6">
        {configs.map((config) => {
          const summary = costSummary[config.period];
          const isOverBudget = summary.percent >= 100;
          const isWarning = summary.percent >= 80 && summary.percent < 100;

          return (
            <div key={config.period} className="border rounded-lg p-4 space-y-4">
              <div className="flex items-center justify-between">
                <h3 className="font-medium">{PERIOD_LABELS[config.period]} 예산</h3>
                {isOverBudget && (
                  <span className="text-xs text-red-600 flex items-center gap-1">
                    <AlertTriangle className="h-3 w-3" /> 예산 초과
                  </span>
                )}
                {isWarning && (
                  <span className="text-xs text-yellow-600 flex items-center gap-1">
                    <AlertTriangle className="h-3 w-3" /> 80% 도달
                  </span>
                )}
              </div>

              {/* Current Usage Progress */}
              {summary.budget && (
                <div className="space-y-1">
                  <div className="flex justify-between text-sm">
                    <span className="text-muted-foreground">현재 사용량</span>
                    <span>
                      ${summary.cost.toFixed(4)} / ${summary.budget.toFixed(2)} ({summary.percent.toFixed(1)}%)
                    </span>
                  </div>
                  <Progress 
                    value={Math.min(summary.percent, 100)} 
                    className={isOverBudget ? '[&>div]:bg-red-500' : isWarning ? '[&>div]:bg-yellow-500' : ''}
                  />
                </div>
              )}

              <div className="grid gap-4 sm:grid-cols-2">
                {/* Budget Amount */}
                <div className="space-y-2">
                  <Label>예산 금액 (USD)</Label>
                  <Input
                    type="number"
                    step="0.01"
                    min="0"
                    value={config.budgetUsd || ''}
                    onChange={(e) => updateConfig(config.period, { 
                      budgetUsd: parseFloat(e.target.value) || 0 
                    })}
                    placeholder="예: 10.00"
                  />
                </div>

                {/* Alert Settings */}
                <div className="space-y-3">
                  <div className="flex items-center justify-between">
                    <Label className="text-sm">80% 도달 시 알림</Label>
                    <Switch
                      checked={config.alertAt80}
                      onCheckedChange={(checked) => updateConfig(config.period, { alertAt80: checked })}
                    />
                  </div>
                  <div className="flex items-center justify-between">
                    <Label className="text-sm">100% 도달 시 알림</Label>
                    <Switch
                      checked={config.alertAt100}
                      onCheckedChange={(checked) => updateConfig(config.period, { alertAt100: checked })}
                    />
                  </div>
                </div>
              </div>

              <Button
                size="sm"
                onClick={() => handleSave(config.period)}
                disabled={savingPeriod === config.period}
              >
                {savingPeriod === config.period ? (
                  <Loader2 className="h-4 w-4 animate-spin" />
                ) : (
                  '저장'
                )}
              </Button>
            </div>
          );
        })}

        <div className="bg-muted p-4 rounded-lg">
          <p className="text-sm text-muted-foreground">
            <strong>알림 정책:</strong> 각 임계값(80%, 100%)에 처음 도달했을 때 1회 알림됩니다. 
            같은 임계값에서 반복 알림은 1시간 후에만 발생합니다.
          </p>
        </div>
      </CardContent>
    </Card>
  );
}

import { optimizeProviderOrder } from './smart-routing';

// 기존 getProviderOrder 함수를 다음으로 교체:
async function getProviderOrder(featureType: FeatureType): Promise<ProviderName[]> {
  const featureConfig = await getFeatureConfig(featureType);
  
  // 사용자 정의 순서가 있으면 그것을 기반으로 최적화
  const customOrder = [
    featureConfig.primaryProvider as ProviderName,
    ...(featureConfig.fallbackOrder as ProviderName[]),
  ];
  
  // 비용 기반 최적화 적용
  return optimizeProviderOrder(featureType, customOrder);
}
