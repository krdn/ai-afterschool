```typescript
import { anthropic } from '@ai-sdk/anthropic';
import { openai } from '@ai-sdk/openai';
import { google } from '@ai-sdk/google';
import { createOllama } from 'ollama-ai-provider-v2';
import type { ProviderName } from './types';

// Ollama 기본 URL (Docker networking용)
const OLLAMA_BASE_URL = process.env.OLLAMA_BASE_URL || 'http://192.168.0.5:11434/api';

// Provider 인스턴스 생성 함수 레지스트리
export const providers = {
  anthropic: (model?: string) => anthropic(model || 'claude-sonnet-4-5'),
  openai: (model?: string) => openai(model || 'gpt-4o'),
  google: (model?: string) => google(model || 'gemini-2.5-flash-preview-05-20'),
  ollama: (model?: string) => {
    const ollamaInstance = createOllama({ baseURL: OLLAMA_BASE_URL });
    return ollamaInstance(model || 'llama3.1:8b');
  },
} as const;

export function getProvider(name: ProviderName, model?: string) {
  const providerFn = providers[name];
  if (!providerFn) {
    throw new Error(`Unknown provider: ${name}`);
  }
  return providerFn(model);
}

export * from './types';
```

1. `src/lib/ai/encryption.ts` 생성:
```typescript
import { createCipheriv, createDecipheriv, randomBytes } from 'crypto';

const ALGORITHM = 'aes-256-gcm';

function getEncryptionKey(): Buffer {
  const key = process.env.API_KEY_ENCRYPTION_SECRET;
  if (!key || key.length !== 64) {
    throw new Error(
      'API_KEY_ENCRYPTION_SECRET must be a 64-character hex string (32 bytes). ' +
      'Generate with: openssl rand -hex 32'
    );
  }
  return Buffer.from(key, 'hex');
}

/**
 * API 키를 암호화합니다.
 * @param plaintext - 평문 API 키
 * @returns 암호화된 문자열 (iv:authTag:encrypted 형식)
 */
export function encryptApiKey(plaintext: string): string {
  const key = getEncryptionKey();
  const iv = randomBytes(16);
  const cipher = createCipheriv(ALGORITHM, key, iv);
  
  let encrypted = cipher.update(plaintext, 'utf8', 'hex');
  encrypted += cipher.final('hex');
  
  const authTag = cipher.getAuthTag();
  
  // Format: iv:authTag:encrypted
  return `${iv.toString('hex')}:${authTag.toString('hex')}:${encrypted}`;
}

/**
 * 암호화된 API 키를 복호화합니다.
 * @param encryptedData - 암호화된 문자열 (iv:authTag:encrypted 형식)
 * @returns 평문 API 키
 */
export function decryptApiKey(encryptedData: string): string {
  const key = getEncryptionKey();
  const [ivHex, authTagHex, encrypted] = encryptedData.split(':');
  
  if (!ivHex || !authTagHex || !encrypted) {
    throw new Error('Invalid encrypted data format');
  }
  
  const iv = Buffer.from(ivHex, 'hex');
  const authTag = Buffer.from(authTagHex, 'hex');
  const decipher = createDecipheriv(ALGORITHM, key, iv);
  decipher.setAuthTag(authTag);
  
  let decrypted = decipher.update(encrypted, 'hex', 'utf8');
  decrypted += decipher.final('utf8');
  
  return decrypted;
}

/**
 * API 키를 마스킹합니다 (UI 표시용).
 * @param apiKey - API 키
 * @returns 마스킹된 문자열 (예: sk-***...***xyz)
 */
export function maskApiKey(apiKey: string): string {
  if (apiKey.length <= 8) {
    return '***';
  }
  const prefix = apiKey.slice(0, 3);
  const suffix = apiKey.slice(-3);
  return `${prefix}***...***${suffix}`;
}
```

```prisma
// ========================================
// LLM Configuration & Usage Tracking
// ========================================

// LLM 제공자 설정 (API 키 등)
model LLMConfig {
  id              String    @id @default(cuid())
  provider        String    @unique // 'anthropic' | 'openai' | 'google' | 'ollama'
  displayName     String    // 'Claude' | 'ChatGPT' | 'Gemini' | 'Ollama'
  apiKeyEncrypted String?   // 암호화된 API 키 (Ollama는 null)
  isEnabled       Boolean   @default(false)
  isValidated     Boolean   @default(false)
  validatedAt     DateTime?
  baseUrl         String?   // Ollama용: 'http://192.168.0.5:11434'
  defaultModel    String?   // 기본 모델 ID
  createdAt       DateTime  @default(now())
  updatedAt       DateTime  @updatedAt
}

// 기능별 LLM 매핑 (스마트 라우팅용)
model LLMFeatureConfig {
  id              String   @id @default(cuid())
  featureType     String   @unique // 'learning_analysis' | 'counseling_suggest' 등
  primaryProvider String   // 우선 제공자
  fallbackOrder   Json     // ['anthropic', 'openai'] 형태
  modelOverride   String?  // 특정 모델 지정 (선택)
  createdAt       DateTime @default(now())
  updatedAt       DateTime @updatedAt
}

// LLM 사용량 추적 (모든 요청 기록)
model LLMUsage {
  id             String   @id @default(cuid())
  provider       String   // 'anthropic' | 'openai' | 'google' | 'ollama'
  modelId        String   // 'claude-sonnet-4-5' | 'gpt-4o' 등
  featureType    String   // 'learning_analysis' | 'counseling_suggest' 등
  teacherId      String?  // 요청한 선생님 ID (선택)
  inputTokens    Int
  outputTokens   Int
  totalTokens    Int
  costUsd        Float    // 계산된 비용 (USD)
  responseTimeMs Int      // 응답 시간 (밀리초)
  success        Boolean  @default(true)
  errorMessage   String?
  failoverFrom   String?  // 폴백인 경우 원래 제공자
  createdAt      DateTime @default(now())

  @@index([provider, createdAt])
  @@index([featureType, createdAt])
  @@index([teacherId, createdAt])
  @@index([createdAt])
}

// 월별 집계 데이터 (영구 보관)
model LLMUsageMonthly {
  id                 String   @id @default(cuid())
  year               Int
  month              Int
  provider           String
  featureType        String
  totalRequests      Int
  totalInputTokens   BigInt
  totalOutputTokens  BigInt
  totalCostUsd       Float
  avgResponseTimeMs  Float
  successRate        Float    // 0-1 범위
  createdAt          DateTime @default(now())

  @@unique([year, month, provider, featureType])
  @@index([year, month])
}

// 비용 예산 및 알림 설정
model LLMBudget {
  id          String    @id @default(cuid())
  period      String    @unique // 'daily' | 'weekly' | 'monthly'
  budgetUsd   Float
  alertAt80   Boolean   @default(true)
  alertAt100  Boolean   @default(true)
  lastAlertAt DateTime? // 마지막 알림 시간 (debounce용)
  lastAlertThreshold Int? // 마지막 알림 임계값 (80 or 100)
  createdAt   DateTime  @default(now())
  updatedAt   DateTime  @updatedAt
}
```
