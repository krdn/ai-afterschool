'use server';

import { verifySession } from '@/lib/session';
import { db } from '@/lib/db';
import { checkAllBudgetStatuses, markAlertSent } from '@/lib/ai/smart-routing';

// 간단한 인메모리 알림 저장소 (프로덕션에서는 DB 모델 사용 권장)
// 서버 재시작 시 초기화됨
const notifications: Array<{
  id: string;
  type: 'cost_alert' | 'info';
  title: string;
  message: string;
  link?: string;
  createdAt: Date;
  read: boolean;
}> = [];

/**
 * 비용 알림을 체크하고 필요시 생성합니다.
 */
export async function checkAndCreateCostAlertsAction() {
  const session = await verifySession();
  if (!session || session.role !== 'DIRECTOR') {
    return { created: 0 };
  }

  const statuses = await checkAllBudgetStatuses();
  let created = 0;

  for (const status of statuses) {
    if (status.shouldAlert && status.alertThreshold) {
      const periodLabel = status.period === 'daily' ? '일일' :
                          status.period === 'weekly' ? '주간' : '월간';
      
      const notification = {
        id: `cost-${status.period}-${status.alertThreshold}-${Date.now()}`,
        type: 'cost_alert' as const,
        title: `${periodLabel} 예산 ${status.alertThreshold}% 도달`,
        message: `현재 $${status.currentCostUsd.toFixed(2)} / $${status.budgetUsd.toFixed(2)} (${status.percentUsed.toFixed(1)}%)`,
        link: '/admin/llm-usage',
        createdAt: new Date(),
        read: false,
      };

      notifications.unshift(notification);
      await markAlertSent(status.period, status.alertThreshold);
      created++;
    }
  }

  // 최대 50개 알림 유지
  while (notifications.length > 50) {
    notifications.pop();
  }

  return { created };
}

/**
 * 알림 목록을 조회합니다.
 */
export async function getNotificationsAction() {
  const session = await verifySession();
  if (!session || session.role !== 'DIRECTOR') {
    return [];
  }

  return notifications.slice(0, 20);
}

/**
 * 알림을 읽음 처리합니다.
 */
export async function markNotificationReadAction(id: string) {
  const session = await verifySession();
  if (!session || session.role !== 'DIRECTOR') {
    return;
  }

  const notification = notifications.find((n) => n.id === id);
  if (notification) {
    notification.read = true;
  }
}

/**
 * 모든 알림을 읽음 처리합니다.
 */
export async function markAllNotificationsReadAction() {
  const session = await verifySession();
  if (!session || session.role !== 'DIRECTOR') {
    return;
  }

  notifications.forEach((n) => {
    n.read = true;
  });
}

/**
 * 읽지 않은 알림 개수를 조회합니다.
 */
export async function getUnreadCountAction() {
  const session = await verifySession();
  if (!session || session.role !== 'DIRECTOR') {
    return 0;
  }

  return notifications.filter((n) => !n.read).length;
}

'use client';

import { useState, useEffect } from 'react';
import { Bell } from 'lucide-react';
import {
  DropdownMenu,
  DropdownMenuContent,
  DropdownMenuItem,
  DropdownMenuLabel,
  DropdownMenuSeparator,
  DropdownMenuTrigger,
} from '@/components/ui/dropdown-menu';
import { Button } from '@/components/ui/button';
import Link from 'next/link';
import {
  getNotificationsAction,
  markNotificationReadAction,
  markAllNotificationsReadAction,
  getUnreadCountAction,
  checkAndCreateCostAlertsAction,
} from '@/lib/actions/notifications';

interface Notification {
  id: string;
  type: 'cost_alert' | 'info';
  title: string;
  message: string;
  link?: string;
  createdAt: Date;
  read: boolean;
}

export function NotificationBell() {
  const [notifications, setNotifications] = useState<Notification[]>([]);
  const [unreadCount, setUnreadCount] = useState(0);
  const [isOpen, setIsOpen] = useState(false);

  // 알림 로드
  const loadNotifications = async () => {
    const [notifs, count] = await Promise.all([
      getNotificationsAction(),
      getUnreadCountAction(),
    ]);
    setNotifications(notifs);
    setUnreadCount(count);
  };

  // 초기 로드 및 주기적 체크
  useEffect(() => {
    loadNotifications();
    
    // 비용 알림 체크 (1분마다)
    const checkAlerts = async () => {
      await checkAndCreateCostAlertsAction();
      await loadNotifications();
    };
    
    const interval = setInterval(checkAlerts, 60000);
    return () => clearInterval(interval);
  }, []);

  // 알림 클릭 시 읽음 처리
  const handleNotificationClick = async (notification: Notification) => {
    if (!notification.read) {
      await markNotificationReadAction(notification.id);
      await loadNotifications();
    }
  };

  // 모두 읽음 처리
  const handleMarkAllRead = async () => {
    await markAllNotificationsReadAction();
    await loadNotifications();
  };

  return (
    <DropdownMenu open={isOpen} onOpenChange={setIsOpen}>
      <DropdownMenuTrigger asChild>
        <Button variant="ghost" size="icon" className="relative">
          <Bell className="h-5 w-5" />
          {unreadCount > 0 && (
            <span className="absolute -top-1 -right-1 h-5 w-5 rounded-full bg-red-500 text-white text-xs flex items-center justify-center">
              {unreadCount > 9 ? '9+' : unreadCount}
            </span>
          )}
        </Button>
      </DropdownMenuTrigger>
      <DropdownMenuContent align="end" className="w-80">
        <DropdownMenuLabel className="flex justify-between items-center">
          <span>알림</span>
          {unreadCount > 0 && (
            <button
              onClick={handleMarkAllRead}
              className="text-xs text-muted-foreground hover:text-foreground"
            >
              모두 읽음
            </button>
          )}
        </DropdownMenuLabel>
        <DropdownMenuSeparator />
        
        {notifications.length === 0 ? (
          <div className="p-4 text-center text-muted-foreground text-sm">
            새 알림이 없습니다
          </div>
        ) : (
          notifications.slice(0, 5).map((notification) => (
            <DropdownMenuItem
              key={notification.id}
              className={`flex flex-col items-start p-3 cursor-pointer ${
                !notification.read ? 'bg-muted/50' : ''
              }`}
              onClick={() => handleNotificationClick(notification)}
              asChild={!!notification.link}
            >
              {notification.link ? (
                <Link href={notification.link}>
                  <div className="font-medium text-sm">{notification.title}</div>
                  <div className="text-xs text-muted-foreground mt-1">
                    {notification.message}
                  </div>
                  <div className="text-xs text-muted-foreground mt-1">
                    {new Date(notification.createdAt).toLocaleString('ko-KR')}
                  </div>
                </Link>
              ) : (
                <div>
                  <div className="font-medium text-sm">{notification.title}</div>
                  <div className="text-xs text-muted-foreground mt-1">
                    {notification.message}
                  </div>
                  <div className="text-xs text-muted-foreground mt-1">
                    {new Date(notification.createdAt).toLocaleString('ko-KR')}
                  </div>
                </div>
              )}
            </DropdownMenuItem>
          ))
        )}
        
        {notifications.length > 5 && (
          <>
            <DropdownMenuSeparator />
            <DropdownMenuItem asChild className="justify-center">
              <Link href="/admin/llm-usage" className="text-sm text-primary">
                모든 알림 보기
              </Link>
            </DropdownMenuItem>
          </>
        )}
      </DropdownMenuContent>
    </DropdownMenu>
  );
}

import { Card, CardContent, CardDescription, CardHeader, CardTitle } from '@/components/ui/card';
import { AlertTriangle, DollarSign, TrendingUp } from 'lucide-react';
import Link from 'next/link';

interface CostAlertsProps {
  costSummary: {
    daily: { cost: number; budget: number | null; percent: number };
    weekly: { cost: number; budget: number | null; percent: number };
    monthly: { cost: number; budget: number | null; percent: number };
  };
}

export function CostAlerts({ costSummary }: CostAlertsProps) {
  const alerts: Array<{
    period: string;
    cost: number;
    budget: number;
    percent: number;
    severity: 'warning' | 'danger';
  }> = [];

  // 경고/위험 상태 수집
  for (const [period, data] of Object.entries(costSummary)) {
    if (data.budget && data.percent >= 80) {
      alerts.push({
        period: period === 'daily' ? '일일' : period === 'weekly' ? '주간' : '월간',
        cost: data.cost,
        budget: data.budget,
        percent: data.percent,
        severity: data.percent >= 100 ? 'danger' : 'warning',
      });
    }
  }

  if (alerts.length === 0) {
    return null;
  }

  return (
    <div className="space-y-4">
      {alerts.map((alert) => (
        <Card 
          key={alert.period} 
          className={`border-l-4 ${
            alert.severity === 'danger' ? 'border-l-red-500 bg-red-50' : 'border-l-yellow-500 bg-yellow-50'
          }`}
        >
          <CardHeader className="pb-2">
            <div className="flex items-center gap-2">
              <AlertTriangle className={`h-5 w-5 ${
                alert.severity === 'danger' ? 'text-red-500' : 'text-yellow-500'
              }`} />
              <CardTitle className="text-base">
                {alert.period} 예산 {alert.severity === 'danger' ? '초과' : '경고'}
              </CardTitle>
            </div>
          </CardHeader>
          <CardContent>
            <div className="flex items-baseline gap-2">
              <span className="text-2xl font-bold">${alert.cost.toFixed(2)}</span>
              <span className="text-muted-foreground">/ ${alert.budget.toFixed(2)}</span>
              <span className={`ml-2 text-sm font-medium ${
                alert.severity === 'danger' ? 'text-red-600' : 'text-yellow-600'
              }`}>
                ({alert.percent.toFixed(1)}%)
              </span>
            </div>
            <Link 
              href="/admin/llm-settings" 
              className="text-sm text-primary hover:underline mt-2 inline-block"
            >
              예산 설정 변경 →
            </Link>
          </CardContent>
        </Card>
      ))}
    </div>
  );
}

import { verifySession } from '@/lib/session';

import { UsageCharts } from './usage-charts';

      endDate: new Date().toISOString(),

  ]);

    <div className="container py-6 space-y-8">
      <div>
        <h1 className="text-2xl font-bold">LLM 사용량</h1>
        <p className="text-muted-foreground">
          AI 서비스 사용량과 비용을 모니터링합니다.
        </p>
      </div>

      {/* Cost Alerts */}
      <CostAlerts costSummary={costSummary} />

      {/* Summary Cards */}
      <div className="grid gap-4 md:grid-cols-4">
        <Card>
          <CardHeader className="flex flex-row items-center justify-between pb-2">
            <CardTitle className="text-sm font-medium">오늘 비용</CardTitle>
            <DollarSign className="h-4 w-4 text-muted-foreground" />
          </CardHeader>
          <CardContent>
            <div className="text-2xl font-bold">${summary.today.totalCostUsd.toFixed(4)}</div>
            <p className="text-xs text-muted-foreground">
              {summary.today.totalRequests.toLocaleString()} 요청
            </p>

            <DollarSign className="h-4 w-4 text-muted-foreground" />
          </CardHeader>
          <CardContent>
            <div className="text-2xl font-bold">${summary.week.totalCostUsd.toFixed(4)}</div>
            <p className="text-xs text-muted-foreground">
              {summary.week.totalRequests.toLocaleString()} 요청
            </p>
          </CardContent>
        </Card>

        <Card>
          <CardHeader className="flex flex-row items-center justify-between pb-2">

          </CardHeader>
          <CardContent>
            <div className="text-2xl font-bold">${summary.month.totalCostUsd.toFixed(2)}</div>
            <p className="text-xs text-muted-foreground">
              {summary.month.totalRequests.toLocaleString()} 요청
            </p>
          </CardContent>
