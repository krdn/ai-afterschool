```typescript
import type { ProviderName } from './providers';

// ============================================================
// Failover Error Types
// ============================================================

export class FailoverError extends Error {
  public readonly failedProviders: Array<{ provider: ProviderName; error: string }>;
  
  constructor(
    message: string,
    failedProviders: Array<{ provider: ProviderName; error: string }>
  ) {
    super(message);
    this.name = 'FailoverError';
    this.failedProviders = failedProviders;
  }
}

// ============================================================
// Error Classification
// ============================================================

export type ErrorType = 
  | 'auth_error'      // API 키 문제 (401, 403)
  | 'rate_limit'      // 요청 한도 초과 (429)
  | 'server_error'    // 서버 에러 (5xx)
  | 'network_error'   // 네트워크 연결 실패
  | 'timeout'         // 타임아웃
  | 'invalid_request' // 잘못된 요청 (4xx)
  | 'unknown';        // 분류 불가

/**
 * 에러 유형을 분류합니다.
 */
export function classifyError(error: unknown): ErrorType {
  if (!(error instanceof Error)) {
    return 'unknown';
  }

  const message = error.message.toLowerCase();

  // 인증 오류
  if (message.includes('401') || message.includes('403') || 
      message.includes('unauthorized') || message.includes('forbidden') ||
      message.includes('invalid api key')) {
    return 'auth_error';
  }

  // 요청 한도 초과
  if (message.includes('429') || message.includes('rate limit') || 
      message.includes('too many requests')) {
    return 'rate_limit';
  }

  // 서버 에러
  if (message.includes('500') || message.includes('502') || 
      message.includes('503') || message.includes('504') ||
      message.includes('internal server error') || message.includes('service unavailable')) {
    return 'server_error';
  }

  // 네트워크 에러
  if (message.includes('econnrefused') || message.includes('enotfound') ||
      message.includes('fetch failed') || message.includes('network')) {
    return 'network_error';
  }

  // 타임아웃
  if (message.includes('timeout') || message.includes('abort')) {
    return 'timeout';
  }

  // 잘못된 요청
  if (message.includes('400') || message.includes('bad request')) {
    return 'invalid_request';
  }

  return 'unknown';
}

/**
 * 폴백이 의미 있는 에러인지 확인합니다.
 * 인증 에러나 잘못된 요청은 다른 제공자도 실패할 가능성이 높으므로 계속 시도.
 */
export function shouldFailover(errorType: ErrorType): boolean {
  // 모든 에러 유형에서 폴백 시도 (CONTEXT.md: 오류 즉시 다음 제공자 시도)
  return true;
}

// ============================================================
// Failover Logging
// ============================================================

interface FailoverLog {
  timestamp: Date;
  fromProvider: ProviderName;
  toProvider: ProviderName;
  errorType: ErrorType;
  errorMessage: string;
  featureType: string;
}

const recentFailovers: FailoverLog[] = [];
const MAX_FAILOVER_LOGS = 100;

/**
 * 폴백 발생을 로깅합니다.
 */
export function logFailover(log: Omit<FailoverLog, 'timestamp'>) {
  const entry: FailoverLog = {
    ...log,
    timestamp: new Date(),
  };

  recentFailovers.unshift(entry);
  
  // 최근 로그만 유지
  if (recentFailovers.length > MAX_FAILOVER_LOGS) {
    recentFailovers.pop();
  }

  // 콘솔에도 경고 출력
  console.warn(
    `[Failover] ${log.fromProvider} → ${log.toProvider} | ` +
    `Error: ${log.errorType} | Feature: ${log.featureType}`
  );
}

/**
 * 최근 폴백 로그를 조회합니다.
 */
export function getRecentFailovers(limit = 10): FailoverLog[] {
  return recentFailovers.slice(0, limit);
}

// ============================================================
// User-Friendly Error Messages
// ============================================================

/**
 * 사용자에게 표시할 에러 메시지를 생성합니다.
 */
export function getUserFriendlyErrorMessage(error: FailoverError): string {
  const failedCount = error.failedProviders.length;
  
  if (failedCount === 0) {
    return 'AI 서비스에 일시적인 문제가 발생했습니다. 잠시 후 다시 시도해주세요.';
  }

  // 모든 에러가 네트워크 관련인 경우
  const allNetwork = error.failedProviders.every(
    (fp) => classifyError(new Error(fp.error)) === 'network_error'
  );
  if (allNetwork) {
    return 'AI 서비스에 연결할 수 없습니다. 네트워크 연결을 확인하고 다시 시도해주세요.';
  }

  // 모든 에러가 인증 관련인 경우
  const allAuth = error.failedProviders.every(
    (fp) => classifyError(new Error(fp.error)) === 'auth_error'
  );
  if (allAuth) {
    return 'AI 서비스 인증에 문제가 있습니다. 관리자에게 문의해주세요.';
  }

  return '모든 AI 제공자에서 오류가 발생했습니다. 잠시 후 다시 시도해주세요.';
}
```

```typescript
export async function generateWithProvider(options: GenerateOptions): Promise<GenerateResult> {
  const { prompt, featureType, teacherId, maxTokens, temperature, system } = options;
  const providerOrder = await getProviderOrder(featureType);
  const featureConfig = await getFeatureConfig(featureType);

  const failedProviders: Array<{ provider: ProviderName; error: string }> = [];
  let failoverFrom: ProviderName | undefined;

  for (let i = 0; i < providerOrder.length; i++) {
    const provider = providerOrder[i];
    const isFailover = i > 0;
    
    if (isFailover) {
      failoverFrom = providerOrder[i - 1];
      
      // 폴백 로깅
      const lastError = failedProviders[failedProviders.length - 1];
      if (lastError) {
        logFailover({
          fromProvider: failoverFrom,
          toProvider: provider,
          errorType: classifyError(new Error(lastError.error)),
          errorMessage: lastError.error,
          featureType,
        });
      }
    }

    const startTime = Date.now();

    try {
      const isReady = await setupProviderEnv(provider);
      if (!isReady) {
        failedProviders.push({ provider, error: 'Provider not configured or enabled' });
        continue;
      }

      const config = await getLLMConfig(provider);
      const modelId = featureConfig.modelOverride || config?.defaultModel || undefined;

      const model = getProvider(provider, modelId);
      const result = await generateText({
        model,
        prompt,
        system,
        maxTokens,
        temperature,
        maxRetries: 0,
      });

      const responseTimeMs = Date.now() - startTime;

      await trackUsage({
        provider,
        modelId: modelId || 'default',
        featureType,
        teacherId,
        inputTokens: result.usage?.inputTokens || 0,
        outputTokens: result.usage?.outputTokens || 0,
        responseTimeMs,
        success: true,
        failoverFrom: isFailover ? failoverFrom : undefined,
      });

      return {
        text: result.text,
        usage: result.usage,
        provider,
        model: modelId || 'default',
        wasFailover: isFailover,
        failoverFrom: isFailover ? failoverFrom : undefined,
      };
    } catch (error) {
      const responseTimeMs = Date.now() - startTime;
      const errorMessage = error instanceof Error ? error.message : 'Unknown error';
      
      failedProviders.push({ provider, error: errorMessage });
      
      await trackFailure({
        provider,
        modelId: featureConfig.modelOverride || 'default',
        featureType,
        teacherId,
        errorMessage,
        responseTimeMs,
      });
    }
  }

  // 모든 제공자 실패
  const failoverError = new FailoverError(
    `All ${failedProviders.length} providers failed`,
    failedProviders
  );
  
  throw failoverError;
}
```
