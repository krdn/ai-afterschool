// ============================================================

/**
 * 특정 월의 사용량을 집계합니다.
 */
export async function aggregateMonthlyUsage(year: number, month: number) {
  // 해당 월의 시작과 끝
  const startDate = new Date(year, month - 1, 1);
  const endDate = new Date(year, month, 0, 23, 59, 59, 999);

  // 제공자 × 기능 조합별로 집계
  const aggregations = await db.lLMUsage.groupBy({
    by: ['provider', 'featureType'],
    where: {
      createdAt: {
        gte: startDate,
        lte: endDate,
      },
    },
    _count: { id: true },
    _sum: {
      inputTokens: true,
      outputTokens: true,
      costUsd: true,
      responseTimeMs: true,
    },
  });

  // 성공률 계산을 위해 성공 건수도 조회
  const successCounts = await db.lLMUsage.groupBy({
    by: ['provider', 'featureType'],
    where: {
      createdAt: {
        gte: startDate,
        lte: endDate,
      },
      success: true,
    },
    _count: { id: true },
  });

  const successMap = new Map(
    successCounts.map((s) => [`${s.provider}-${s.featureType}`, s._count.id])
  );

  // 각 집계 결과를 월별 테이블에 upsert
  const results = await Promise.all(
    aggregations.map(async (agg) => {
      const key = `${agg.provider}-${agg.featureType}`;
      const successCount = successMap.get(key) || 0;
      const totalCount = agg._count.id;

      return db.lLMUsageMonthly.upsert({
        where: {
          year_month_provider_featureType: {
            year,
            month,
            provider: agg.provider,
            featureType: agg.featureType,
          },
        },
        create: {
          year,
          month,
          provider: agg.provider,
          featureType: agg.featureType,
          totalRequests: totalCount,
          totalInputTokens: BigInt(agg._sum.inputTokens || 0),
          totalOutputTokens: BigInt(agg._sum.outputTokens || 0),
          totalCostUsd: agg._sum.costUsd || 0,
          avgResponseTimeMs: totalCount > 0 
            ? (agg._sum.responseTimeMs || 0) / totalCount 
            : 0,
          successRate: totalCount > 0 ? successCount / totalCount : 1,
        },
        update: {
          totalRequests: totalCount,
          totalInputTokens: BigInt(agg._sum.inputTokens || 0),
          totalOutputTokens: BigInt(agg._sum.outputTokens || 0),
          totalCostUsd: agg._sum.costUsd || 0,
          avgResponseTimeMs: totalCount > 0 
            ? (agg._sum.responseTimeMs || 0) / totalCount 
            : 0,
          successRate: totalCount > 0 ? successCount / totalCount : 1,
        },
      });
    })
  );

  return {
    year,
    month,
    aggregatedCount: results.length,
  };
}

/**
 * 이전 달의 사용량을 집계합니다 (크론 작업용).
 */
export async function aggregatePreviousMonth() {
  const now = new Date();
  const prevMonth = now.getMonth(); // 0-indexed, 현재 달 - 1
  const year = prevMonth === 0 ? now.getFullYear() - 1 : now.getFullYear();
  const month = prevMonth === 0 ? 12 : prevMonth;

  return aggregateMonthlyUsage(year, month);
}

export async function cleanupOldUsageData() {

      createdAt: {

    },
  });

  };
}

/**
 * 월별 사용량 통계를 조회합니다.
 */
export async function getMonthlyStats(options: {
  startYear: number;
  startMonth: number;
  endYear: number;
  endMonth: number;
  provider?: ProviderName;
  featureType?: FeatureType;
}) {
  const { startYear, startMonth, endYear, endMonth, provider, featureType } = options;

  const where: {
    OR: Array<{
      year: number;
      month: { gte?: number; lte?: number };
    }>;
    provider?: string;
    featureType?: string;
  } = {
    OR: [],
  };

  // 기간 조건 구성
  if (startYear === endYear) {
    where.OR.push({
      year: startYear,
      month: { gte: startMonth, lte: endMonth },
    });
  } else {
    // 시작 년도
    where.OR.push({
      year: startYear,
      month: { gte: startMonth },
    });
    // 중간 년도들
    for (let y = startYear + 1; y < endYear; y++) {
      where.OR.push({
        year: y,
        month: { gte: 1, lte: 12 },
      });
    }
    // 끝 년도
    where.OR.push({
      year: endYear,
      month: { lte: endMonth },
    });
  }

  if (provider) where.provider = provider;
  if (featureType) where.featureType = featureType;

  return db.lLMUsageMonthly.findMany({
    where,
    orderBy: [{ year: 'asc' }, { month: 'asc' }],
  });
}

/**
 * 일별 사용량을 조회합니다 (차트용).
 */
export async function getDailyUsage(options: {
  startDate: Date;
  endDate: Date;
  provider?: ProviderName;
}): Promise<Array<{
  date: string;
  requests: number;
  inputTokens: number;
  outputTokens: number;
  costUsd: number;
}>> {
  const { startDate, endDate, provider } = options;

  const raw = await db.lLMUsage.findMany({
    where: {
      createdAt: {
        gte: startDate,
        lte: endDate,
      },
      ...(provider && { provider }),
    },
    select: {
      createdAt: true,
      inputTokens: true,
      outputTokens: true,
      costUsd: true,
    },
    orderBy: { createdAt: 'asc' },
  });

  // 일별로 그룹화
  const dailyMap = new Map<string, {
    requests: number;
    inputTokens: number;
    outputTokens: number;
    costUsd: number;
  }>();

  for (const record of raw) {
    const dateKey = record.createdAt.toISOString().split('T')[0];
    const existing = dailyMap.get(dateKey) || {
      requests: 0,
      inputTokens: 0,
      outputTokens: 0,
      costUsd: 0,
    };

    existing.requests += 1;
    existing.inputTokens += record.inputTokens;
    existing.outputTokens += record.outputTokens;
    existing.costUsd += record.costUsd;

    dailyMap.set(dateKey, existing);
  }

  // 배열로 변환
  return Array.from(dailyMap.entries())
    .map(([date, stats]) => ({
      date,
      ...stats,
    }))
    .sort((a, b) => a.date.localeCompare(b.date));
}

async function requireDirector() {
  const session = await verifySession();
  if (!session || session.role !== 'DIRECTOR') {
    throw new Error('Unauthorized: DIRECTOR role required');
  }
  return session;
}

import { NextResponse } from 'next/server';
import { aggregatePreviousMonth, cleanupOldUsageData } from '@/lib/ai/usage-aggregation';

/**
 * 월별 LLM 사용량 집계 Cron 엔드포인트
 * 
 * 매월 1일 00:05에 실행 (이전 달 집계)
 * vercel.json에서 설정하거나 외부 cron 서비스에서 호출
 * 
 * vercel.json 예시:
 * {
 *   "crons": [{
 *     "path": "/api/cron/aggregate-llm-usage",
 *     "schedule": "5 0 1 * *"
 *   }]
 * }
 */
export async function GET(request: Request) {
  // Cron 인증 (선택적)
  const authHeader = request.headers.get('Authorization');
  const cronSecret = process.env.CRON_SECRET;
  
  if (cronSecret && authHeader !== `Bearer ${cronSecret}`) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
  }

  try {
    // 1. 이전 달 집계
    const aggregationResult = await aggregatePreviousMonth();

    // 2. 오래된 상세 데이터 정리 (1년 이상)
    const cleanupResult = await cleanupOldUsageData();

    console.log('[Cron] LLM Usage Aggregation completed:', {
      aggregation: aggregationResult,
      cleanup: cleanupResult,
    });

    return NextResponse.json({
      success: true,
      aggregation: aggregationResult,
      cleanup: cleanupResult,
      timestamp: new Date().toISOString(),
    });
  } catch (error) {
    console.error('[Cron] LLM Usage Aggregation failed:', error);
    
    return NextResponse.json({
      success: false,
      error: error instanceof Error ? error.message : 'Unknown error',
    }, { status: 500 });
  }
}

// POST도 허용 (외부 cron 서비스 호환)
export async function POST(request: Request) {
  return GET(request);
}
