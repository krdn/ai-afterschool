    ```typescript
    'use server'

    import { verifySession } from '@/lib/dal'
    import { existsSync, readdirSync, statSync } from 'fs'
    import { join } from 'path'

    export interface BackupFileEntry {
      name: string
      size: number
      createdAt: Date
      path: string
    }

    export async function getBackupList(): Promise<BackupFileEntry[]> {
      const session = await verifySession()

      // 권한 검증: 원장만 접근 가능
      if (session.role !== 'DIRECTOR') {
        return []
      }

      const backupDir = process.env.BACKUP_DIR || './backups'
      const dbName = process.env.POSTGRES_DB || 'ai_afterschool'

      // 백업 디렉토리 확인
      if (!existsSync(backupDir)) {
        return []
      }

      try {
        // 백업 파일 목록 가져오기
        const files = readdirSync(backupDir)
          .filter((f) => f.startsWith(`${dbName}-`) && f.endsWith('.sql.gz'))
          .map((f) => {
            const filePath = join(backupDir, f)
            const stats = statSync(filePath)
            return {
              name: f,
              size: stats.size,
              createdAt: stats.mtime,
              path: filePath,
            }
          })
          .sort((a, b) => b.createdAt.getTime() - a.createdAt.getTime()) // 최신 순

        return files
      } catch (error) {
        console.error('Failed to read backup directory:', error)
        return []
      }
    }
    ```

    ```typescript
    export const metadata = {
      title: '데이터베이스 백업 | AI AfterSchool',
      description: '데이터베이스 백업 관리',
    }

    function formatBytes(bytes: number): string {
      if (bytes === 0) return '0 Bytes'
      const k = 1024
      const sizes = ['Bytes', 'KB', 'MB', 'GB']
      const i = Math.floor(Math.log(bytes) / Math.log(k))
      return Math.round(bytes / Math.pow(k, i) * 100) / 100 + ' ' + sizes[i]
    }

    export default async function DatabaseBackupPage() {
      const session = await verifySession()
      if (!session || session.role !== 'DIRECTOR') {
        redirect('/dashboard')
      }

      const backups = await getBackupList()

      return (
        <div className="container py-6 space-y-8">
          <div className="flex items-center justify-between">
            <div>
              <h1 className="text-2xl font-bold">데이터베이스 백업</h1>
              <p className="text-gray-500">
                {backups.length > 0
                  ? `총 ${backups.length}개의 백업 파일이 있어요`
                  : '백업 파일이 없어요'}
              </p>
            </div>
          </div>

          {/* 백업 안내 */}
          <div className="bg-blue-50 border border-blue-200 rounded-lg p-4">
            <p className="text-sm text-blue-800">
              <strong>백업 정보:</strong> 백업 파일은{' '}
              <code className="bg-blue-100 px-1 rounded">
                {process.env.BACKUP_DIR || './backups'}
              </code>
              디렉토리에 저장됩니다. 자동 백업은 cron으로 설정되어 있어야 합니다.
            </p>
          </div>

          {/* 백업 목록 */}
          <div data-testid="backup-list">
            {backups.length === 0 ? (
              <div className="text-center py-12 bg-gray-50 border rounded-lg" data-testid="no-backups">
                <p className="text-gray-500 mb-2">백업 파일이 없어요</p>
                <p className="text-sm text-gray-400">
                  자동 백업이 설정되어 있지 않거나 아직 백업이 실행되지 않았습니다.
                </p>
              </div>
            ) : (
              <div className="border rounded-lg">
                <Table>
                  <TableHeader>
                    <TableRow>
                      <TableHead>파일명</TableHead>
                      <TableHead>크기</TableHead>
                      <TableHead>생성일</TableHead>
                      <TableHead>관리</TableHead>
                    </TableRow>
                  </TableHeader>
                  <TableBody>
                    {backups.map((backup) => (
                      <TableRow key={backup.name} data-testid="backup-row">
                        <TableCell
                          className="font-mono text-sm"
                          data-testid="backup-name"
                        >
                          {backup.name}
                        </TableCell>
                        <TableCell data-testid="backup-size">
                          {formatBytes(backup.size)}
                        </TableCell>
                        <TableCell data-testid="backup-date">
                          {backup.createdAt.toLocaleString('ko-KR')}
                        </TableCell>
                        <TableCell>
                          <Button
                            variant="ghost"
                            size="sm"
                            asChild
                            disabled
                            data-testid="download-backup-button"
                          >
                            <span className="flex items-center gap-1 opacity-50">
                              <Download className="w-4 h-4" />
                              다운로드
                            </span>
                          </Button>
                        </TableCell>
                      </TableRow>
                    ))}
                  </TableBody>
                </Table>
              </div>
            )}

          {backups.length > 0 && (
            <div className="text-sm text-gray-500">
              <p>백업 파일 위치: <code>{process.env.BACKUP_DIR || './backups'}</code></p>
            </div>
          )}
        </div>
      )
    }
    ```
