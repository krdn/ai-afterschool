       ```typescript
         async function fetchAnalyticsData() {
           setLoading(true)
           setError(null)
           try {
             const [teachersRes, trendRes, comparisonRes, counselingRes] = await Promise.all([
               getTeacherStudentMetrics(),
               getGradeTrendDataAction(),
               compareTeachersByGradeImprovement(),
               getCounselingStats()
             ])

             setTeachers(teachersRes)
             setGradeTrendData(trendRes)
             setComparisonData(comparisonRes)
             setCounselingStats(counselingRes)
           } catch (err) {
             console.error('Failed to fetch analytics data:', err)
             setError('데이터를 불러오는데 실패했습니다')
           } finally {
             setLoading(false)
           }
         }

         fetchAnalyticsData()
       }, [])
       ```
