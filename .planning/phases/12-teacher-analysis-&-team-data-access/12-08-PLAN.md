   ```typescript
   export async function runTeacherPalmAnalysis(
     teacherId: string,
     imageUrl: string,
     hand: "left" | "right"
   ) {
     const session = await verifySession()
     if (!session) throw new Error("Unauthorized")

     // AI 손금 분석 실행 (기존 함수 재사용)
     const result = await analyzePalmImage(imageUrl, hand)

     // DB 저장
     await upsertTeacherPalmAnalysis(teacherId, {
       hand,
       imageUrl,
       result: result as any,
       status: "complete",
       version: 1,
       analyzedAt: new Date()
     })

     revalidatePath(`/teachers/${teacherId}`)

     return { success: true }
   }
   ```
