#!/bin/bash
set -e

# AI AfterSchool Zero-Downtime Deployment Script
#
# Usage:
#   ./scripts/deploy.sh                    # Interactive mode
#   ./scripts/deploy.sh --force            # Skip confirmation
#   ./scripts/deploy.sh --tag=v1.1.0       # Deploy specific version
#   ./scripts/deploy.sh --skip-health      # Skip health check (not recommended)
#
# Environment variables:
#   COMPOSE_FILE: Path to docker-compose file (default: docker-compose.prod.yml)
#   HEALTH_CHECK_TIMEOUT: Seconds to wait for health check (default: 60)
#   HEALTH_CHECK_URL: URL to check (default: http://localhost:3000/api/health)

# Configuration
COMPOSE_FILE="${COMPOSE_FILE:-docker-compose.prod.yml}"
HEALTH_CHECK_TIMEOUT="${HEALTH_CHECK_TIMEOUT:-60}"
HEALTH_CHECK_URL="${HEALTH_CHECK_URL:-http://localhost:3000/api/health}"
BACKUP_TAG="backup-$(date +%Y%m%d-%H%M%S)"
FORCE_DEPLOY=false
SKIP_HEALTH=false

# Colors for output
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
NC='\033[0m' # No Color

# Functions
log_info() {
    echo -e "${BLUE}[INFO]${NC} $1"
}

log_success() {
    echo -e "${GREEN}[SUCCESS]${NC} $1"
}

log_warning() {
    echo -e "${YELLOW}[WARNING]${NC} $1"
}

log_error() {
    echo -e "${RED}[ERROR]${NC} $1"
}

print_banner() {
    echo "═════════════════════════════════════════════════════════"
    echo "  AI AfterSchool - Zero-Downtime Deployment"
    echo "═════════════════════════════════════════════════════════"
    echo ""
}

parse_arguments() {
    while [[ $# -gt 0 ]]; do
        case $1 in
            --force)
                FORCE_DEPLOY=true
                shift
                ;;
            --tag=*)
                VERSION_TAG="${1#*=}"
                shift
                ;;
            --skip-health)
                SKIP_HEALTH=true
                shift
                ;;
            -h|--help)
                echo "Usage: $0 [options]"
                echo ""
                echo "Options:"
                echo "  --force              Skip confirmation prompt"
                echo "  --tag=VERSION        Deploy specific version"
                echo "  --skip-health        Skip health check"
                echo "  -h, --help           Show this help"
                exit 0
                ;;
            *)
                log_error "Unknown option: $1"
                exit 1
                ;;
        esac
    done
}

confirm_deployment() {
    if [ "$FORCE_DEPLOY" = true ]; then
        return 0
    fi

    echo ""
    read -p "Do you want to proceed with deployment? (yes/no): " -r
    echo ""
    if [[ ! $REPLY =~ ^[Yy][Ee][Ss]$ ]]; then
        log_info "Deployment cancelled by user"
        exit 0
    fi
}

check_prerequisites() {
    log_info "Checking prerequisites..."

    # Check if docker-compose is available
    if ! command -v docker-compose &> /dev/null; then
        log_error "docker-compose not found"
        exit 1
    fi

    # Check if compose file exists
    if [ ! -f "$COMPOSE_FILE" ]; then
        log_error "Compose file not found: $COMPOSE_FILE"
        exit 1
    fi

    # Check if .env exists
    if [ ! -f .env ] && [ ! -f .env.production ]; then
        log_warning "No .env file found. Using .env.production template..."
        if [ -f .env.production ]; then
            cp .env.production .env
            log_warning "Created .env from .env.production. Please review values!"
        else
            log_error "No environment configuration found"
            exit 1
        fi
    fi

    log_success "Prerequisites check passed"
}

backup_current_version() {
    log_info "Backing up current version..."

    # Tag current running images with backup tag
    docker-compose -f "$COMPOSE_FILE" ps -q | \
        xargs -I {} sh -c 'docker inspect {} --format="{{.Config.Image}}"' 2>/dev/null | \
        sort -u | \
        while read image; do
            if [ -n "$image" ] && [ "$image" != "" ]; then
                docker tag "$image" "${image}:${BACKUP_TAG}" 2>/dev/null || true
                log_info "Backed up: $image -> ${image}:${BACKUP_TAG}"
            fi
        done

    log_success "Backup completed with tag: $BACKUP_TAG"
}

build_new_version() {
    log_info "Building new version..."

    docker-compose -f "$COMPOSE_FILE" build --no-cache app

    log_success "Build completed"
}

deploy_new_version() {
    log_info "Deploying new version..."

    # Pull latest images
    docker-compose -f "$COMPOSE_FILE" pull

    # Start new containers (graceful shutdown of old ones)
    docker-compose -f "$COMPOSE_FILE" up -d --no-deps --build app

    log_success "New containers started"
}

wait_for_health_check() {
    if [ "$SKIP_HEALTH" = true ]; then
        log_warning "Skipping health check (--skip-health flag used)"
        return 0
    fi

    log_info "Waiting for application to be healthy..."
    log_info "Timeout: ${HEALTH_CHECK_TIMEOUT}s"

    local elapsed=0
    local interval=5

    while [ $elapsed -lt $HEALTH_CHECK_TIMEOUT ]; do
        if curl -f -s "$HEALTH_CHECK_URL" > /dev/null 2>&1; then
            local response=$(curl -s "$HEALTH_CHECK_URL")
            local status=$(echo "$response" | grep -o '"status":"[^"]*"' | cut -d'"' -f4)

            if [ "$status" = "healthy" ]; then
                log_success "Health check passed! (took ${elapsed}s)"
                return 0
            fi
        fi

        echo -n "."
        sleep $interval
        elapsed=$((elapsed + interval))
    done

    echo ""
    log_error "Health check failed after ${HEALTH_CHECK_TIMEOUT}s"

    # Automatic rollback
    log_warning "Initiating automatic rollback..."
    rollback_deployment
    exit 1
}

restart_dependent_services() {
    log_info "Restarting dependent services..."

    # Restart Caddy to pick up new app
    docker-compose -f "$COMPOSE_FILE" restart caddy

    log_success "Services restarted"
}

cleanup_old_images() {
    log_info "Cleaning up old images (keeping last 5 versions)..."

    # Remove dangling images
    docker image prune -f

    log_success "Cleanup completed"
}

rollback_deployment() {
    log_error "Rolling back to previous version..."

    # Stop current containers
    docker-compose -f "$COMPOSE_FILE" stop app

    # Restart with backup tag
    # Note: This requires docker-compose to support image tags in service definition
    # For now, we'll just restart and let it use the previous image
    docker-compose -f "$COMPOSE_FILE" up -d app

    log_warning "Rollback completed. Please investigate the failure."
}

show_deployment_summary() {
    echo ""
    echo "═════════════════════════════════════════════════════════"
    echo "  Deployment Summary"
    echo "═════════════════════════════════════════════════════════"
    echo ""
    echo "Status: ${GREEN}SUCCESS${NC}"
    echo "Backup Tag: $BACKUP_TAG"
    echo ""
    echo "Running containers:"
    docker-compose -f "$COMPOSE_FILE" ps
    echo ""
    echo "═════════════════════════════════════════════════════════"
}

# Main deployment flow
main() {
    print_banner
    parse_arguments "$@"
    check_prerequisites
    confirm_deployment

    echo ""
    log_info "Starting deployment process..."
    echo ""

    backup_current_version
    build_new_version
    deploy_new_version
    wait_for_health_check
    restart_dependent_services
    cleanup_old_images

    show_deployment_summary
}

# Run main function
main "$@"

#!/bin/bash
set -e

# AI AfterSchool Rollback Script
#
# Usage:
#   ./scripts/rollback.sh                      # Rollback to last backup
#   ./scripts/rollback.sh --tag=backup-20250130-120000  # Rollback to specific tag

COMPOSE_FILE="${COMPOSE_FILE:-docker-compose.prod.yml}"
TARGET_TAG=""

# Colors
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
NC='\033[0m'

log_info() {
    echo -e "${BLUE}[INFO]${NC} $1"
}

log_success() {
    echo -e "${GREEN}[SUCCESS]${NC} $1"
}

log_error() {
    echo -e "${RED}[ERROR]${NC} $1"
}

print_banner() {
    echo "═════════════════════════════════════════════════════════"
    echo "  AI AfterSchool - Rollback"
    echo "═════════════════════════════════════════════════════════"
    echo ""
}

parse_arguments() {
    while [[ $# -gt 0 ]]; do
        case $1 in
            --tag=*)
                TARGET_TAG="${1#*=}"
                shift
                ;;
            -h|--help)
                echo "Usage: $0 [options]"
                echo ""
                echo "Options:"
                echo "  --tag=TAG    Rollback to specific backup tag"
                echo "  -h, --help   Show this help"
                exit 0
                ;;
            *)
                log_error "Unknown option: $1"
                exit 1
                ;;
        esac
    done
}

list_backup_tags() {
    log_info "Available backup tags:"
    echo ""
    docker images --format "{{.Repository}}:{{.Tag}}" | \
        grep -E "backup-[0-9]{8}-[0-9]{6}" | \
        sort -r | \
        head -10
    echo ""
}

confirm_rollback() {
    echo ""
    read -p "Do you want to proceed with rollback? (yes/no): " -r
    echo ""
    if [[ ! $REPLY =~ ^[Yy][Ee][Ss]$ ]]; then
        log_info "Rollback cancelled by user"
        exit 0
    fi
}

perform_rollback() {
    log_info "Stopping current containers..."
    docker-compose -f "$COMPOSE_FILE" stop app

    log_info "Starting rollback container..."
    # Note: This is a simplified rollback
    # In production, you'd want to tag images and restore by tag
    docker-compose -f "$COMPOSE_FILE" up -d app

    log_success "Rollback completed"
}

main() {
    print_banner
    parse_arguments "$@"

    if [ -z "$TARGET_TAG" ]; then
        list_backup_tags
        log_info "No specific tag provided. Will restart last known good version."
    else
        log_info "Rolling back to: $TARGET_TAG"
    fi

    confirm_rollback
    perform_rollback

    echo ""
    log_info "Current containers:"
    docker-compose -f "$COMPOSE_FILE" ps
    echo ""
    log_success "Rollback complete. Please verify application health."
}

main "$@"

## Troubleshooting

### Health Check Timeout

**Symptom:** Deployment fails with "Health check failed"

**Solutions:**
1. Check application logs: `docker-compose logs app`
2. Check database connectivity: `docker-compose logs postgres`
3. Manual health check: `curl http://localhost:3000/api/health`
4. Skip health check for emergency: `npm run deploy:skip-health`

### Container Won't Start

**Symptom:** App container exits immediately

**Solutions:**
1. Check logs: `docker-compose logs app`
2. Verify environment variables in `.env`
3. Check database connection: `docker-compose exec postgres psql -U postgres`
4. Verify MinIO connectivity: `curl http://localhost:9000/minio/health/live`

### Rollback Required

**Symptom:** New deployment broken, need to revert

**Solution:**
```bash
npm run rollback
```

Or manual rollback:
```bash
docker-compose -f docker-compose.prod.yml down
docker-compose -f docker-compose.prod.yml up -d
```
